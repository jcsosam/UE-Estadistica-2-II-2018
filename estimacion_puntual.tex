\begin{frame}{Estimadores}

\begin{block}{Objetivo}
	\begin{enumerate}
		\item Estamos interesados en un parámetro $\theta$ que toma valores en el espacio de parámetros $\Theta$.
		\item Dado un modelo estadístico $\{F_\theta : \theta \in \Theta\}$, ¿cuál es una \textbf{aproximación razonable} para el parámetro $\theta$ basada en una muestra aleatoria $X_1,\ldots,X_n$ de la población?
	\end{enumerate}
		 
\end{block}

\begin{block}{Estimador (estadístico)}
	Un \textbf{estimador} del parámetro $\theta$ es una función de la muestra aleatoria $\hat{\theta} = \hat{\theta}(X_1,\ldots,X_n)$  que toma valores en el espacio de parámetros $\Theta$.
\end{block}

\begin{block}{Estimador vs. Estimación}
	    \textbf{Los estimadores son variables aleatorias} (sobre las cuales es posible calcular probabilidades), mientras que \textbf{las estimaciones son realizaciones} particulares de los estimadores.
\end{block}

\end{frame}



\begin{frame}{Estimadores de Máxima Verosimilitud (Fisher)}

\begin{block}{Función de verosimilitud}
	Sea $\muestra$ una muestra aleatoria de una población $X$ con función de densidad (o masa) $f_X(x;\theta)$. La \textbf{función de verosimilitud} de $\muestra$ es
	\vspace{-8pt}
	$$ L(\theta) = \prod_{i=1}^n f_X(x_i;\theta) $$
	\vspace{-10pt}
	
	La función de verosimilitud no es más que la \textbf{función de densidad (o masa) conjunta}
	de una muestra aleatoria, pero entendida como una
	\textbf{función del parámetro dadas las observaciones}.
\end{block}


\begin{block}{Estimador de Máxima Verosimilitud (MLE)}
	El \textbf{estimador de máxima verosimilitud} de $\theta$, denotado con $\thetahat_{\mle}$, maximiza la función de verosimilitud, es decir,
	$$
	\thetahat_{\mle} = \argmax_{\theta\in \Theta} L(\theta)
	$$
\end{block}

\begin{block}{Log-verosimilitud}
	La log-verosimilitud es el logaritmo (natural) de la función de verosimilitud: \\
	\centering	$ \ell(\theta) = \log L(\theta) $
\end{block}

\end{frame}


\begin{frame}{Propiedades de estimadores}

\begin{block}{Estadístco Suficiente (definición informal)}
	Si $\log L(\theta)$ depende de $x_1,\ldots,x_n$ unicamente a través de $t =T(x_1,\ldots,x_n)$, se dice que el estadístico
	$T = T(\muestra)$ es un \textbf{estadístico suficiente} para $\theta$.\\
	\vspace{5pt}
	Si queremos decir algo sobre un
	el parámetro $\theta$ y la estadística $T$ es suficiente para $\theta$, entonces reportar la realización
	de $T$ da tanta información acerca de $\theta$ como la realización de la muestra completa.
\end{block}

\begin{block}{Estimador Insesgado}
	$\thetahat$ es un \textbf{estimador insesgado} de $\theta$ si $\mathbb{E}[\thetahat] = \theta$.
\end{block}


\begin{block}{Observación}
	Existen más cracterísticas de los estimadores:
	\begin{itemize}
		\item Consistencia.
		\item Eficiencia.
		\item Completez.
	\end{itemize}
\end{block}

\end{frame}


\begin{frame}{Estimadores de Máxima Verosimilitud (cont.)}

\begin{block}{Información Observada de Fisher}
La información observada de Fisher se define como
$$
\hat{I} = -\frac{\partial^2}{\partial\theta^2}\,\ell(\theta) \Big|_{\theta = \thetahat_{\mle}}
$$
\end{block}


\begin{block}{Teorema (distribución asintótica del MLE)}
	Sea $\muestra$ una muestra aleatoria de una población $X$
	cuya distribución depende de un parámetro desconocido $\theta$.
	Si el tamaño de la muestra es \quo{grande},
	entonces bajo algunas \textit{condiciones de regularidad} se tiene que
	\vspace{-3pt}
	$$
	\thetahat_{\mle} \sima \Nor\left(\theta,\hat{I}^{-1}\right)
	$$
\end{block}

\begin{block}{Intervalo de confianza}
Un \textbf{intervalo de confianza para $\bs{\thetahat_{\mle}}$ usando una confiabilidad del $\bs{100(1-\alpha)\%}$} es
\vspace{-5pt}
$$
\thetahat_{\mle} \pm z_{1-\alpha/2}\,\sqrt{\hat{I}^{-1}}
$$
\vspace{-15pt}

donde $\thetahat_{\mle}$ es el MLE de $\theta$, $\hat{I}$ es la información observada de Fisher, y $z_{1-\alpha/2}$ es el percentíl $100(1-\alpha/2)$ de la distribución Normal estándar.
\end{block}


\end{frame}
