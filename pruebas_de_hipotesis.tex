\section{Introducción}

\begin{frame}{Terminología}

\begin{block}{Hipótesis estadística}
\textbf{Aseveración} (afirmación) acerca de uno o más \textbf{parámetros} poblacionales.
\begin{itemize}
  \item Hipótesis \textbf{simple}: $H:\mu = \mu_0$.
  \item Hipótesis \textbf{compuesta}: $H:\mu \geq \mu_0$, $H:\mu \leq \mu_0$, $H:\mu > \mu_0$, $H:\mu < \mu_0$, $H:\mu \neq \mu_0$.
  \item $\mu_0$ se denomina \textbf{valor hipotético}.
\end{itemize}
\end{block}

\begin{block}{Sistema de hipótesis}
\textbf{Arreglo de dos hipótesis} denominas \textbf{hipótesis nula} e \textbf{hipótesis alternativa}.
\begin{itemize}
  \item $H_0:\mu \leq \mu_0$ frente a $H_1:\mu > \mu_0$.
  \item $H_0:\mu \geq \mu_0$ frente a $H_1:\mu < \mu_0$.
  \item $H_0:\mu = \mu_0$ frente a $H_1:\mu \neq \mu_0$.
\end{itemize}
\end{block}

\begin{block}{Objetivo}
Establecer si hay \textbf{suficiente evidencia} en la realización de una muestra aleatoria para \textbf{rechazar} o \textbf{no rechazar} la \textbf{hipótesis nula}. ¿Cuál hipótesis explica mejor los datos?
\end{block}

\end{frame}



\begin{frame}{Terminología}

\begin{block}{Tipos de error}
\begin{itemize}
  \item Error tipo 1: \textbf{Rechazar} la hipótesis nula cuando ésta es cierta.
  \item Error tipo 2: \textbf{No rechazar} la hipótesis nula cuando ésta es falsa.
\end{itemize}
\end{block}

\begin{figure}[h!]
\centering
\includegraphics[scale=.25]{./figs/errores.pdf}
\end{figure}

\begin{block}{Nivel de significativa}
\begin{itemize}
  \item \textbf{Probabilidad} de cometer el \textbf{error tipo 1}: $$\alpha = \pr{\text{Error tipo 1}} = \pr{\text{Rechazar $H_0$}\mid H_0}$$
  \item El nivel de significativa usualmente \textbf{se fija de antemano} a 0.05 o 0.01.
\end{itemize}
\end{block}

\end{frame}



\begin{frame}{Terminología}

\begin{block}{Analogía}
$$H_0:\text{El acusado es \textbf{inocente}}\quad\text{frente a}\quad H_1:\text{El acusado es \textbf{culpable}}$$

El acusado se considera inocente hasta que se demuestre lo contrario más allá de cualquier duda razonable.
\end{block}


\begin{block}{Rechazar $H_0$}
Se ha encontrado algo en la muestra tan improbable que ocurra si $H_0$ es cierta, que obliga al investigador a favorecer $H_1$ (rechazar $H_0$).
\end{block}

\begin{block}{La ciencia es conservadora!}
\begin{itemize}
  \item Se considera mejor, equivocadamente no rechazar, que equivocadamente rechazar el estado actual del conocimiento ($H_0$).
  \item $H_0$ se mantiene a menos de que haya suficiente evidencia para revocarla.
\end{itemize}
\end{block}

\end{frame}



\begin{frame}{Procedimiento de prueba (test)}

\begin{block}{¿Cómo decidir si se debe rechazar o no la hipótesis nula?}
\begin{enumerate}[1.]
  \item Establecer el \textbf{sistema de hipótesis} de interés y el \textbf{nivel de significancia}.
  \item Calcular el \textbf{estadístico de prueba}.
  \item Establecer la \textbf{región crítica} o el \textbf{valor} $\boldsymbol{p}$.
  \item Tomar la \textbf{decisión}.
  \item {\color{blue} \textbf{Interpretar los resultados!}}
\end{enumerate}
\end{block}


\begin{block}{¿Cómo tomar la decisión?}

Si el estadístico de prueba pertenece a la región crítica, entonces se rechaza $H_0$.

Si el valor $p$ es menor que $\alpha$, entonces se rechaza $H_0$. 
\end{block}

\end{frame}


\begin{frame}{Pruebas de hipótesis para la media poblacional}

\begin{block}{Modelo}
$$X_1,X_2,\ldots,X_n\simiid \Nor(\mu, \sigma^2)$$
\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\mu\leq\mu_0\quad\text{frente a}\quad H_1:\mu > \mu_0 $\\
$ H_0:\mu\geq\mu_0\quad\text{frente a}\quad H_1:\mu < \mu_0 $\\
$ H_0:\mu=\mu_0   \quad\text{frente a}\quad H_1:\mu \neq \mu_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
t=\frac{\bar{X} - \mu_0}{S/\sqrt{n}}\sim \tdistr_{n-1}
$$
Nota: Para $n\geq 30$ se tiene que $\tdistr_{n-1}\approx \Nor(0,1)$ y la población no tiene que ser Normal.
\end{block}

\begin{block}{Región de rechazo}
$$
\text{Rechazar $H_0$ si }
\left\{
  \begin{array}{ll}
     t  \,\,\,> \tdistr_{1-\alpha,n-1}, & \hbox{si $H_1:\mu > \mu_0$;} \\
     t  \,\,\,< \tdistr_{\alpha,n-1},   & \hbox{si $H_1:\mu < \mu_0$;} \\
    |t| > \tdistr_{1-\alpha/2,n-1},     & \hbox{si $H_1:\mu \neq \mu_0$.}
  \end{array}
\right.
$$
\end{block}

\end{frame}



\begin{frame}{Valor $p$}

\begin{block}{Definición}
  $$p = \pr{\text{Observar datos \textbf{tan o más extremos} en dirección de $H_1$}\mid H_0}$$
Probabilidad de que el azar produzca una \textbf{coincidencia al menos tan extraordinaria} como el fenómeno observado, bajo la hipótesis nula.
\end{block}

\begin{block}{Observación}
Rechazar $H_0$ siempre que $p <\alpha$.

\vspace{3pt}
Se ha observado algo que es \textbf{tan poco probable que ocurra} si la hipótesis nula es cierta que hace dudar seriamente que realmente sea cierta.
\end{block}

\begin{block}{Calculo del valor $p$ para la media poblacional}
  $$
  p =
\left\{
  \begin{array}{ll}
     \pr{\tdistr_{n-1} > t},       & \hbox{si $H_1:\mu > \mu_0$;} \\
     \pr{\tdistr_{n-1} < t},       & \hbox{si $H_1:\mu < \mu_0$;} \\
     2\,\pr{\tdistr_{n-1} > |t|},  & \hbox{si $H_1:\mu \neq \mu_0$.}
  \end{array}
\right.
  $$

\vspace{3pt}
Nota: Para $n\geq 30$ se tiene que $\tdistr_{n-1}\approx \Nor(0,1)$.
\end{block}

\end{frame}



\begin{frame}{Ejemplo}

\begin{block}{Pruebas de hipótesis para la media poblacional}
	Con propósitos de acreditación institucional, se quiere comprobar si la media de las calificaciones promedio de todos los estudiantes de una Facultad difiere de 4.0. \\
	\vspace{5pt}
	
	Para tal fin, se toma una muestra aleatoria de estudiantes de la Facultad y se registra la calificación promedio del semestre inmediatamente anterior de cada estudiante (ver los datos correspondientes en \texttt{PROM.txt}).\\
	\vspace{5pt}
	
	Usando un nivel de significancia de 5\%, probar el sistema de hipótesis correspondiente.
\end{block}

\end{frame}



\begin{frame}{Pruebas de hipótesis para la proporción poblacional}

\begin{block}{Modelo}
	$$X_1,X_2,\ldots,X_n\simiid \Ber(\pi)$$
\end{block}

\begin{block}{Sistema de hipótesis}
	\centering
	$ H_0:\pi\leq\pi_0\quad\text{frente a}\quad H_1:\pi > \pi_0 $\\
	$ H_0:\pi\geq\pi_0\quad\text{frente a}\quad H_1:\pi < \pi_0 $\\
	$ H_0:\pi=\pi_0   \quad\text{frente a}\quad H_1:\pi \neq \pi_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
	$$
	z=\frac{\hat{\pi} - \pi_0}{\sqrt{\frac{\pi_0(1-\pi_0)}{n}}}\approx \Nor(0,1)
	$$
	Nota: Esta aproximación es apropiada para $n\geq30$, $n\hat{\pi} \geq 5$ y $n(1-\hat{\pi}) \geq 5$.
\end{block}

\begin{block}{Región de rechazo}
	$$
	\text{Rechazar $H_0$ si }
	\left\{
	\begin{array}{ll}
	z  \,\,\,> z_{1-\alpha}, & \hbox{si $H_1:\pi > \pi_0$;} \\
	z  \,\,\,< z_{\alpha},   & \hbox{si $H_1:\pi < \pi_0$;} \\
	|z| > z_{1-\alpha/2},     & \hbox{si $H_1:\pi \neq \pi_0$.}
	\end{array}
	\right.
	$$
\end{block}

\end{frame}



\begin{frame}{Ejemplo}

\begin{block}{Pruebas de hipótesis para la proporción poblacional}
	Hacer trampa ha sido una preocupación del decano de la Facultad de Negocios de Bayview University
	por muchos años. Algunos miembros de la Facultad de negocios en la Universidad creen que hacer trampa es más
	extendido en Bayview que en otras universidades, mientras que otros miembros de la Facultad piensan que
	hacer trampa no es un problema importante en la universidad. Para resolver algunos de estos problemas, el decano
	encargó un estudio para evaluar el comportamiento ético actual de los estudiantes de negocios en Bayview.
	Como parte de este estudio, se administró una encuesta de salida anónima a una muestra de 90 
	estudiantes de la clase de graduación de 2009. Las respuestas a las preguntas están dadas en \texttt{BAYVIEW.txt}
	en relación a tres tipos de trampas.\\
	
	\vspace{5pt}
	

Realizar una prueba de hipótesis para determinar si la proporción de estudiantes de negocios en
		La Universidad Bayview que estuvo involucrada en algún tipo de trampa es menor que la de
		estudiantes de negocios en otras instituciones según lo informado por \textit{Chronicle of Higher
			Eudcation} (el artículo asegura que 56\% de los estudiantes de negocios hacen trampa en algún momento de su carrera académica).
		

\end{block}

\end{frame}



\begin{frame}{Pruebas de hipótesis para la varianza poblacional}

\begin{block}{Modelo}
$$X_1,X_2,\ldots,X_n\simiid \Nor(\mu, \sigma^2)$$
\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\sigma^2\leq\sigma^2_0\quad\text{frente a}\quad H_1:\sigma^2 > \sigma^2_0 $\\
$ H_0:\sigma^2\geq\sigma^2_0\quad\text{frente a}\quad H_1:\sigma^2 < \sigma^2_0 $\\
$ H_0:\sigma^2=\sigma^2_0   \quad\text{frente a}\quad H_1:\sigma^2 \neq \sigma^2_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
\chi^2=\frac{(n-1)S^2}{\sigma^2_0}\sim \chi^2_{n-1}
$$
\end{block}

\begin{block}{Región de rechazo}
$$
\text{Rechazar $H_0$ si }
\left\{
  \begin{array}{ll}
     \chi^2  >  \chi^2_{1-\alpha,n-1}, & \hbox{si $H_1:\sigma^2 > \sigma^2_0$;} \\
     \chi^2  <  \chi^2_{\alpha,n-1},   & \hbox{si $H_1:\sigma^2 < \sigma^2_0$;} \\
      \chi^2 <  \chi^2_{\alpha/2,n-1}\text{ o }\chi^2 >  \chi^2_{1-\alpha/2,n-1},     & \hbox{si $H_1:\sigma^2 \neq \sigma^2_0$.}
  \end{array}
\right.
$$
\end{block}

\end{frame}



\begin{frame}{Ejemplo}

\begin{block}{Pruebas de hipótesis para la varianza poblacional}
A fines de 2008, la variación en los rendimientos semestrales de bonos gubernamentales en el extranjero era
$\sigma^2=0.70$. Un grupo de inversores se reunió en ese momento para analizar las tendencias futuras de
rendimientos en el exterior. Algunos esperaban que la variabilidad en los rendimientos de los bonos en el extranjero aumentara y otros
tomaron la posición opuesta. Los datos dados en\texttt{ YIELDS.txt} corresponden a los rendimientos semestrales de 12 países desde el 6 de marzo de 2009 (\textit{Barron's}, 9 de marzo de 2009).\\
\vspace{5pt}

Desarrollar y probar la hipótesis para comprobar si los datos indican que la variabilidad en los
los rendimientos han cambiado desde finales de 2008. Usar un nivel de significancia del 5\%.

\end{block}

\end{frame}







\begin{frame}{Pruebas de hipótesis para la diferencia de proporciones poblacional}

\begin{block}{Modelo}
Poblaciones independientes:
\vspace{-5pt}
$$X_1,X_2,\ldots,X_{n_1}\simiid \Ber(\pi_1)\qquad Y_1,Y_2,\ldots,Y_{n_2}\simiid \Ber(\pi_2)$$

\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\pi_1-\pi_2\leq \delta_0 \quad\text{frente a}\quad H_1:\pi_1-\pi_2 > \delta_0 $\\
$ H_0:\pi_1-\pi_2\geq \delta_0 \quad\text{frente a}\quad H_1:\pi_1-\pi_2 < \delta_0 $\\
$ H_0:\pi_1-\pi_2 =   \delta_0 \quad\text{frente a}\quad H_1:\pi_1-\pi_2 \neq \delta_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
z=\frac{\hat{\pi}_1 - \hat{\pi}_2 - \delta_0}{\sqrt{ \hat{\pi}(1-\hat{\pi})\left(\frac{1}{n_1} + \frac{1}{n_2}\right) }}\approx \Nor(0,1)\qquad\hat{\pi} = \frac{x_1 + x_2}{n_1+n_2}
$$
Nota: Esta aproximación es apropiada para
$$n_1,n_2\geq30 \quad n_1\hat{\pi}_1,n_1\hat{\pi}_2 \geq 5 \quad n_1(1-\hat{\pi}_1), n_2(1-\hat{\pi}_2) \geq 5$$
\end{block}

\end{frame}



\begin{frame}{Ejemplo}

\begin{block}{Pruebas de hipótesis para la diferencia de proporciones poblacional}

Para comparar la aceptación de una medida económica en dos universidades se toma una muestra aleatoria en cada una obteniéndose los dados en \texttt{ESTEC.txt}. 
\vspace{5pt}

Probar si hay diferencias significativas entre la proporción de estudiantes que aprueban la medida. Usar un nivel de significancia de 5\%.


\end{block}

\end{frame}



\begin{frame}{Pruebas de hipótesis para el cociente de varianzas poblacional}

\begin{block}{Modelo}
Poblaciones independientes:
\vspace{-5pt}
$$X_1,X_2,\ldots,X_{n_1}\simiid \Nor(\mu_1,\sigma^2_1)\qquad Y_1,Y_2,\ldots,Y_{n_2}\simiid \Nor(\mu_2,\sigma^2_2)$$

\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\sigma^2_1/\sigma^2_2\leq 1 \quad\text{frente a}\quad H_1:\sigma^2_1/\sigma^2_2 > 1 $\\
$ H_0:\sigma^2_1/\sigma^2_2\geq 1 \quad\text{frente a}\quad H_1:\sigma^2_1/\sigma^2_2 < 1 $\\
$ H_0:\sigma^2_1/\sigma^2_2 =   1 \quad\text{frente a}\quad H_1:\sigma^2_1/\sigma^2_2 \neq 1 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
f=\frac{s_1^2}{s_2^2}\sim F_{n_1-1,n_2-1}
$$
Nota: Esta prueba es indispensable antes de realizar una prueba de hipótesis sobre la diferencia de medias.

\end{block}

\end{frame}


\begin{frame}{Pruebas de hipótesis para la diferencia de medias poblacional}

\begin{block}{Modelo}
	Poblaciones independientes:
	\vspace{-5pt}
$$X_1,X_2,\ldots,X_{n_1}\simiid \Nor(\mu_1,\sigma^2_1)\qquad Y_1,Y_2,\ldots,Y_{n_2}\simiid \Nor(\mu_2,\sigma^2_2)$$

\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\mu_1-\mu_2\leq \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 > \delta_0 $\\
$ H_0:\mu_1-\mu_2\geq \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 < \delta_0 $\\
$ H_0:\mu_1-\mu_2 =   \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 \neq \delta_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
t=\frac{\bar{x_1} - \bar{x}_2 - \delta_0}{s_p\sqrt{\frac{1}{n_1} + \frac{1}{n_2}}}\sim t_{n_1+n_2-2}
$$
Nota: Este estadístico de prueba está construido bajo el supuesto de homogeneidad ($\sigma^2_1=\sigma^2_2$)

\end{block}

\end{frame}



\begin{frame}{Pruebas de hipótesis para la diferencia de medias poblacional (cont.)}

\begin{block}{Modelo}
		Poblaciones independientes:
	\vspace{-5pt}
$$X_1,X_2,\ldots,X_{n_1}\simiid \Nor(\mu_1,\sigma^2_1)\qquad Y_1,Y_2,\ldots,Y_{n_2}\simiid \Nor(\mu_2,\sigma^2_2)$$

\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\mu_1-\mu_2\leq \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 > \delta_0 $\\
$ H_0:\mu_1-\mu_2\geq \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 < \delta_0 $\\
$ H_0:\mu_1-\mu_2 =   \delta_0 \quad\text{frente a}\quad H_1:\mu_1-\mu_2 \neq \delta_0 $
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
t=\frac{\bar{x_1} - \bar{x}_2 - \delta_0}{\sqrt{\frac{s^2_1}{n_1} + \frac{s^2_2}{n_2}}}\sim t_{\nu}
\qquad
\nu = \frac{\left(\frac{s^2_1}{n_1} + \frac{s^2_2}{n_2}\right)^2}{\frac{1}{n_1-1}\left(\frac{s^2_1}{n_1}\right)^2 + \frac{1}{n_2-1}\left(\frac{s^2_2}{n_2}\right)^2}
$$
Nota: Este estadístico de prueba está construido bajo el supuesto de heterogeneidad ($\sigma^2_1\neq\sigma^2_2$)
\end{block}

\end{frame}



\begin{frame}{Ejemplo}

Millones de inversores compran fondos de inversión, eligiendo entre miles de posibilidades. 
Algunos fondos se pueden comprar directamente desde bancos u otras instituciones financieras, mientras que otros deben comprarse a través de intermediarios,

¿Es posible concluir con un nivel de significancia del 5\% que los fondos comprados directamente superan

\end{block}

\end{frame}




\begin{frame}{Prueba chi cuadrado de bondad de ajuste}

\begin{block}{Experimento multinomial}
	\begin{itemize}
		\item Número fijo $n$ de repeticiones.
		\item El resultado de cada ensayo se puede clasificar en una de las $k$ categorías.
		\item La probabilidad $\pi_i$ de que el resultado caerá en la categoría $i$ permanece constante
		para cada prueba. Además, $\pi_1+\pi_2+\ldots+\pi_k = 1$.
		\item Cada ensayo del experimento es independiente de los otros ensayos.
	\end{itemize}
\end{block}

\begin{block}{Sistema de hipótesis}
\centering
$ H_0:\pi_1 = \pi_1^0,\ldots,\pi_k = \pi_k^0$\\
frente a\\
$H_1:\text{ al menos un $\pi_i$ no es igual al valor especificado}$
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
$$
\chi^2 = \sum_{i=1}^{k}\frac{(f_i-e_i)^2}{e_i}\sim\chi^2_{k-1}
$$
donde las $f_i$ y las $e_i = n\pi_i^0$ son las frecuencias (absolutas) observadas y esperadas, respectivamente. 
\end{block}


\end{frame}

\begin{frame}{Prueba chi cuadrado de independencia}


\begin{block}{Sistema de hipótesis}
	\centering
	$ H_0:$ Las variables son independientes\\
	frente a\\
	$H_1:$ Las variables son dependientes
\end{block}

\begin{block}{Estadístico de prueba y distribución probabilística de referencia}
	$$
	\chi^2 = \sum_{i=1}^{f}\sum_{j=1}^{c}\frac{(f_{ij}-e_{ij})^2}{e_{ij}}\sim\chi^2_{(f-1)(c-1)}
	$$
	donde las $f_{ij}$ y las $e_{ij} = n_{i\cdot}\,n_{\cdot j}/n$ son las frecuencias (absolutas) observadas y esperadas, respectivamente. 
\end{block}


\end{frame}
